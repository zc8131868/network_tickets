Method 1: Extract from Network Requests (Most Reliable)

This method captures sy-cinfo from actual network requests made by BPM.

Method 2: Extract from Cookies & Storage

This method tries to extract tokens from cookies and browser storage.

Method 3: Manual Copy Helper

Run this script in the BPM page's console (F12 ‚Üí Console tab):

// Copy this entire script and paste it in BPM page console (F12)
(function() {
    const tokens = {
        SY_ACCESS_TOKEN: null,
        SY_UID: null,
        'sy-cinfo': null
    };
    
    // Try cookies
    document.cookie.split(';').forEach(c => {
        const [name, value] = c.trim().split('=');
        if (name === 'SY_ACCESS_TOKEN') tokens.SY_ACCESS_TOKEN = value;
        if (name === 'SY_UID') tokens.SY_UID = value;
        if (name === 'sy-cinfo' || name === 'SY_CINFO') tokens['sy-cinfo'] = value;
    });
    
    // Try localStorage
    try {
        if (!tokens.SY_ACCESS_TOKEN) tokens.SY_ACCESS_TOKEN = localStorage.getItem('SY_ACCESS_TOKEN');
        if (!tokens.SY_UID) tokens.SY_UID = localStorage.getItem('SY_UID');
        if (!tokens['sy-cinfo']) tokens['sy-cinfo'] = localStorage.getItem('sy-cinfo') || localStorage.getItem('SY_CINFO');
    } catch(e) {}
    
    // Try sessionStorage
    try {
        if (!tokens.SY_ACCESS_TOKEN) tokens.SY_ACCESS_TOKEN = sessionStorage.getItem('SY_ACCESS_TOKEN');
        if (!tokens.SY_UID) tokens.SY_UID = sessionStorage.getItem('SY_UID');
        if (!tokens['sy-cinfo']) tokens['sy-cinfo'] = sessionStorage.getItem('sy-cinfo') || sessionStorage.getItem('SY_CINFO');
    } catch(e) {}
    
    // Display results
    console.log('=== Extracted Tokens ===');
    console.log('SY_ACCESS_TOKEN:', tokens.SY_ACCESS_TOKEN ? tokens.SY_ACCESS_TOKEN.substring(0, 50) + '...' : 'NOT FOUND');
    console.log('SY_UID:', tokens.SY_UID || 'NOT FOUND');
    console.log('sy-cinfo:', tokens['sy-cinfo'] || 'NOT FOUND');
    
    // Copy to clipboard
    const text = `SY_ACCESS_TOKEN: ${tokens.SY_ACCESS_TOKEN || 'NOT FOUND'}\nSY_UID: ${tokens.SY_UID || 'NOT FOUND'}\nsy-cinfo: ${tokens['sy-cinfo'] || 'NOT FOUND'}`;
    navigator.clipboard.writeText(text).then(() => {
        console.log('‚úì Tokens copied to clipboard!');
        alert('Tokens copied to clipboard! Check console for details.');
    }).catch(() => {
        console.log('Could not copy to clipboard. Copy manually from above.');
    });
    
    return tokens;
})();
